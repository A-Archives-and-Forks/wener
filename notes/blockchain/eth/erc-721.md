## 获取所有 Token

```js
const myContract = new web3.eth.Contract(abiJson, contractAddress);
myContract
  .getPastEvents('Transfer', {
    filter: {
      _from: '0x0000000000000000000000000000000000000000',
    },
    fromBlock: 0,
  })
  .then((events) => {
    for (let event of events) {
      console.log(event.returnValues._tokenId);
    }
  });
```
